@vspace{2ex}

Translate the Circles of Evaluation into Code.

@do{
  (define expr1 '(+ 4 -9))
  (define expr2 '(- -1 14))
  (define expr3 '(* -5 6.2))
  (define expr4 '(/ 0 (* 12 -9)))
  (define expr5 '(* (+ 9.1 -1) 7))
  (define expr6 '(+ 3 (* 8 -8)))
  (define expr7 '(- (* 6 -2) (+ 0 1)))
  (define expr8 '(+ -13 (/ 100 (+ 3 6))))
  (define expr9 '(- (* (- 15 7) 7) 6.5))
}
