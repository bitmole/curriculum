@do{(define exprs '((- (* 5 9) (+ 2 7))
                 (* (+ 3 12) 16)
                 (+ 20 (* -4 6))
                 (/ 15 (+ 10 (* 4 -2)))
                 (+ (/ 20 (+ 2 3)) (- 6 3))
                 (* 4 (- 8 (+ 2 3)))
                 (- (+ 4 2) (/ 10 5))
                 (+ (- 4 5) (* 5 8))
                 (/ -15 (+ 5 -8))
                 (* (+ 4 (- 9 8)) 5)
                 ))}

@do{(define exprs-as-code
(map sexp-to-code  exprs))}

@do{(define exprs-as-math
(map sexp-to-math  exprs))}
